function openEvent(id){
  currentEvent = EVENTS.find(e=>e.id===id);
  if(!currentEvent) return;

  document.getElementById('modalTitle').textContent = currentEvent.name;
  document.getElementById('modalSub').textContent = `${currentEvent.day}/${currentEvent.month}/${currentEvent.year} — ${currentEvent.location}`;

  const strip = document.getElementById('modalInfoStrip');
  strip.innerHTML = `
    <span style="font-size:12px;color:var(--muted);display:flex;align-items:center;gap:4px"><i class="ti ti-calendar" style="font-size:13px"></i> ${currentEvent.day}/${currentEvent.month}/${currentEvent.year}</span>
    <span style="font-size:12px;color:var(--muted);display:flex;align-items:center;gap:4px"><i class="ti ti-map-pin" style="font-size:13px"></i> ${currentEvent.location}</span>
    <span style="font-size:12px;color:var(--muted);display:flex;align-items:center;gap:4px"><i class="ti ti-users" style="font-size:13px"></i> ${currentEvent.members} người tham gia</span>
    <span style="font-size:12px;color:var(--muted);display:flex;align-items:center;gap:4px"><i class="ti ti-chart-bar" style="font-size:13px"></i> Tiến độ: ${currentEvent.progress}%</span>
  `;

  const allTabs = ['phan-cong','ke-hoach','tinh-nguyen'];
  allTabs.forEach(t=>{
    const el = document.getElementById('tab-'+t);
    const allowed = currentEvent.allowedTabs.includes(t);
    el.style.display = 'flex';
  });

  switchTab('info');
  document.getElementById('modalOverlay').classList.add('open');
}

function switchTab(name){
  ['info','phan-cong','ke-hoach','tinh-nguyen'].forEach(t=>{
    document.getElementById('tab-'+t)?.classList.remove('active');
    document.getElementById('panel-'+t)?.classList.remove('active');
  });
  document.getElementById('tab-'+name)?.classList.add('active');
  document.getElementById('panel-'+name)?.classList.add('active');

  if(name==='info') renderInfoTab();
  else if(name==='phan-cong') renderPhanCongTab();
  else if(name==='ke-hoach') renderKeHoachTab();
  else if(name==='tinh-nguyen') renderTinhNguyenTab();
}

function showAccessDenied(){

function addFiles(input, listId, areaId){
  const list = document.getElementById(listId);
  Array.from(input.files).forEach(f=>{
    const size = f.size<1024*1024?(f.size/1024).toFixed(0)+'KB':(f.size/1024/1024).toFixed(1)+'MB';
    const div = document.createElement('div');
    div.className='uploaded-file';
    div.innerHTML=`<i class="ti ti-file-text"></i><div class="file-meta"><div class="file-name">${f.name}</div><div class="file-size">${size}</div></div><i class="ti ti-x file-del" onclick="this.parentElement.remove()"></i>`;
    list.appendChild(div);
  });
}

function handleDrop(e, prefix){
  e.preventDefault();
  document.getElementById(prefix+'-area').classList.remove('drag');
  const fakeInput = {files:e.dataTransfer.files};
  addFiles(fakeInput, prefix+'-files', prefix+'-area');
}

function submitForm(listId, notifId){
  const list = document.getElementById(listId);
  const notif = document.getElementById(notifId);
  if(!list || list.children.length===0){alert('Vui lòng chọn ít nhất một file.');return;}
  notif.style.display='flex';
  list.innerHTML='';
  setTimeout(()=>notif.style.display='none', 5000);
}
